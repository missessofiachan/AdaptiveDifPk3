    override void WorldTick()
    {
        if (!IsEngineActive()) return;
        if (level.time < 5 || !playeringame[consoleplayer]) return;
        PlayerInfo p = players[consoleplayer];
        if (!p || !p.mo) return;

        CVar presetCv = CVar.FindCVar('adaptive_active_preset');
        if (presetCv && presetCv.GetInt() != lastObservedPreset) ApplyPresetValues();
        
        if (level.time % 35 == 0)
        {
            int maxTotalAmmo = 0;
            int currentTotalAmmo = 0;
            for (Inventory item = p.mo.Inv; item != null; item = item.Inv)
            {
                let ammoItem = Ammo(item);
                if (ammoItem)
                {
                    currentTotalAmmo += ammoItem.Amount;
                    maxTotalAmmo += ammoItem.MaxAmount;
                }
            }
            isAmmoStarved = (maxTotalAmmo > 0 && (double(currentTotalAmmo) / double(maxTotalAmmo)) < 0.15);
            if (killChainCounter > 0) killChainCounter--;

            CVar maxCv = CVar.FindCVar('adaptive_max_difficulty');
            CVar minCv = CVar.FindCVar('adaptive_min_difficulty');
            double maxDiff = maxCv ? maxCv.GetFloat() : 2.0;
            double minDiff = minCv ? minCv.GetFloat() : 0.5;

            if (minDiff != maxDiff)
            {
                double diffDelta = 0.0;
                let basicArmor = BasicArmor(p.mo.FindInventory("BasicArmor"));
                int armorAmount = basicArmor ? basicArmor.Amount : 0;
                int totalHealthPool = p.mo.health + armorAmount;
                CVar aggressiveCv = CVar.FindCVar('adaptive_aggressive_tracking');
                double scaleFactor = (aggressiveCv && aggressiveCv.GetBool()) ? 2.0 : 1.0;
                
                CVar rampCv = CVar.FindCVar('adaptive_ramp_up_speed');
                CVar coolCv = CVar.FindCVar('adaptive_cool_down_speed');
                double rampUp = rampCv ? rampCv.GetFloat() : 0.03;
                double coolDown = coolCv ? coolCv.GetFloat() : 0.05;

                if (totalHealthPool > 160)      diffDelta = rampUp * 1.5 * scaleFactor;
                else if (p.mo.health >= 80)     diffDelta = rampUp * scaleFactor;
                else if (p.mo.health < 35)      diffDelta = -coolDown * scaleFactor;
                currentDifficulty = Clamp(currentDifficulty + diffDelta, minDiff, maxDiff);
            }
        }

        CVar allowRespawnCv = CVar.FindCVar('adaptive_allow_respawn');
        if (allowRespawnCv && allowRespawnCv.GetBool()) ProcessRespawnQueue(p.mo.health);

        CVar itemRecycleCv = CVar.FindCVar('adaptive_item_respawn');
        if (itemRecycleCv && itemRecycleCv.GetBool()) ProcessItemRecycleQueue();
    }

    // Unified helper core logic for uncapped multiplication scaling routines
    void ExecuteCloningCore(Actor mon, bool bIsARespawn)
    {
        CVar extraSpawnsCv = CVar.FindCVar('adaptive_allow_extra_spawns');
        if (!extraSpawnsCv || !extraSpawnsCv.GetBool()) return;

        // Route structural evaluations strictly against specific preference sliders
        CVar baseToggle = CVar.FindCVar('adaptive_clone_base_spawns');
        CVar respawnToggle = CVar.FindCVar('adaptive_clone_respawn_spawns');
        
        if (!bIsARespawn && baseToggle && !baseToggle.GetBool()) return;
        if (bIsARespawn && respawnToggle && !respawnToggle.GetBool()) return;

        PlayerInfo p = players[consoleplayer];
        if (p && p.mo)
        {
            CVar maxSpRangeCv = CVar.FindCVar('adaptive_spawn_max_range');
            double maxSpRange = maxSpRangeCv ? maxSpRangeCv.GetFloat() : 2048.0;
            if ((mon.pos - p.mo.pos).Length() > maxSpRange) return;
        }

        CVar ammoProtCv = CVar.FindCVar('adaptive_ammo_protection');
        if (ammoProtCv && ammoProtCv.GetBool() && isAmmoStarved) return;

        mon.GiveInventory("AdaptiveSpawnToken", 1);
        
        CVar chanceCv = CVar.FindCVar('adaptive_spawn_chance_mult');
        double baseChance = chanceCv ? chanceCv.GetFloat() : 25.0;
        
        double scaledProbability = baseChance;
        CVar realTimeCv = CVar.FindCVar('adaptive_spawn_realtime');
        if (realTimeCv && realTimeCv.GetBool())
        {
            scaledProbability = baseChance * currentDifficulty;
        }
        
        CVar aggChainsCv = CVar.FindCVar('adaptive_aggression_chains');
        if (aggChainsCv && aggChainsCv.GetBool() && killChainCounter >= 8) 
        {
            scaledProbability *= 1.5;
        }

        int clonesToSpawn = int(scaledProbability / 100.0);
        double remainderChance = scaledProbability % 100.0;
        
        if (FRandom[AdaptiveSpawnGen](0.0, 100.0) <= remainderChance) clonesToSpawn++;
        clonesToSpawn = Min(clonesToSpawn, 150); // Hard protection cap expanded to safely prevent stack overflows

        for (int i = 0; i < clonesToSpawn; i++)
        {
            Vector3 spawnPos = mon.pos + (FRandom[AdaptiveOffset](-96, 96), FRandom[AdaptiveOffset](-96, 96), 0);
            Actor clone = Actor.Spawn(mon.GetClass(), spawnPos, ALLOW_REPLACE);
            
            if (clone)
            {
                if (!clone.TestMobjLocation()) 
                {
                    clone.Destroy();
                }
                else
                {
                    clone.GiveInventory("AdaptiveSpawnToken", 1);
                    clone.bAMBUSH = mon.bAMBUSH;
                }
            }
        }
    }

    override void WorldThingSpawned(WorldEvent e)
    {
        if (!IsEngineActive()) return;
        if (e.Thing && e.Thing.bMISSILE && !e.Thing.bISMONSTER)
        {
            CVar projCv = CVar.FindCVar('adaptive_ai_projectiles');
            if (projCv && projCv.GetBool() && currentDifficulty > 1.2 && e.Thing.target && e.Thing.target.bISMONSTER && !e.Thing.target.player)
            {
                e.Thing.Speed = e.Thing.default.Speed * currentDifficulty;
                e.Thing.Vel *= currentDifficulty; 
            }
            return;
        }

        let mapItem = Inventory(e.Thing);
        if (mapItem && !mapItem.Owner && !mapItem.bISMONSTER && !mapItem.player)
        {
            bool itemKnown = false;
            for (int i = 0; i < mapItemRegistry.Size(); i++)
            {
                if (mapItemRegistry[i].spawnPos == mapItem.pos && mapItemRegistry[i].className == mapItem.GetClassName())
                {
                    itemKnown = true;
                    mapItemRegistry[i].pickedUp = false; 
                    break;
                }
            }
            if (!itemKnown)
            {
                let regRecord = new("TrackedMapItem");
                regRecord.className = mapItem.GetClassName();
                regRecord.spawnPos = mapItem.pos;
                regRecord.pickedUp = false;
                regRecord.respawnTimer = 0;
                mapItemRegistry.Push(regRecord);
            }
            return;
        }

        let mon = Actor(e.Thing);
        if (!mon || !mon.bISMONSTER || mon.player || mon.health <= 0) return;
        if (mon.FindInventory("AdaptiveSpawnToken")) return;

        CVar healthMultCv = CVar.FindCVar('adaptive_health_mult');
        if (healthMultCv && healthMultCv.GetBool()) 
            mon.health = int(mon.default.health * currentDifficulty);

        CVar speedMultCv = CVar.FindCVar('adaptive_speed_mult');
        if (speedMultCv && speedMultCv.GetBool()) 
            mon.speed = mon.default.speed * (0.5 + (currentDifficulty * 0.5));

        CVar nmFlagsCv = CVar.FindCVar('adaptive_nightmare_flags');
        if (nmFlagsCv && nmFlagsCv.GetBool())
        {
            if (currentDifficulty >= 1.4) mon.bMISSILEMORE = true;
            if (currentDifficulty >= 1.9) mon.bALWAYSFAST = true;
        }

        CVar painResistCv = CVar.FindCVar('adaptive_ai_pain_resist');
        if (painResistCv && painResistCv.GetBool() && currentDifficulty > 1.0)
        {
            mon.PainChance = Max(0, int(mon.default.PainChance * (2.0 - currentDifficulty)));
        }

        CVar reflexesCv = CVar.FindCVar('adaptive_ai_reflexes');
        if (reflexesCv && reflexesCv.GetBool() && currentDifficulty >= 1.4)
        {
            mon.ReactionTime = Max(0, int(mon.default.ReactionTime / currentDifficulty));
        }

        // Fresh Level Spawn Intercept (Passed as false since it is a raw map load)
        ExecuteCloningCore(mon, false);
    }

    override void WorldThingDestroyed(WorldEvent e)
    {
        if (!IsEngineActive()) return;
        let itemRef = Inventory(e.Thing);
        if (itemRef && !itemRef.Owner)
        {
            CVar recCv = CVar.FindCVar('adaptive_item_respawn');
            if (recCv && recCv.GetBool())
            {
                if (Weapon(itemRef)) 
                {
                    CVar wpCv = CVar.FindCVar('adaptive_item_respawn_weapons');
                    if (!wpCv || !wpCv.GetBool()) return;
                }

                for (int i = 0; i < mapItemRegistry.Size(); i++)
                {
                    if (!mapItemRegistry[i].pickedUp && mapItemRegistry[i].spawnPos == itemRef.pos && mapItemRegistry[i].className == itemRef.GetClassName())
                    {
                        mapItemRegistry[i].pickedUp = true;
                        CVar timeCv = CVar.FindCVar('adaptive_item_respawn_time');
                        double sDelay = timeCv ? timeCv.GetFloat() : 30.0;
                        sDelay *= currentDifficulty;
                        mapItemRegistry[i].respawnTimer = int(35 * Max(1.0, sDelay));
                        break;
                    }
                }
            }
        }
    }

    override void WorldThingDamaged(WorldEvent e)
    {
        if (!IsEngineActive()) return;
        if (!e.Thing || !e.DamageSource || e.Thing.health <= 0) return;
        
        CVar damageMultCv = CVar.FindCVar('adaptive_damage_mult');
        if (damageMultCv && damageMultCv.GetBool() && e.DamageSource.bISMONSTER && e.Thing.player)
        {
            int originalDamage = e.Damage;
            int adaptedDamage = int(originalDamage * currentDifficulty);
            int damageDelta = adaptedDamage - originalDamage;
            
            if (damageDelta != 0)
            {
                e.Thing.health -= damageDelta;
            }
        }

        CVar infightCv = CVar.FindCVar('adaptive_infight_mitigation');
        if (infightCv && infightCv.GetBool() && currentDifficulty >= 1.4)
        {
            if (e.Thing.bISMONSTER && e.DamageSource.bISMONSTER && e.Thing != e.DamageSource)
            {
                int mitigationHeal = int(e.Damage * 0.5);
                e.Thing.health += mitigationHeal;
            }
        }
    }

    override void WorldThingDied(WorldEvent e)
    {
        if (!IsEngineActive()) return;
        let mon = Actor(e.Thing);
        if (mon && mon.bISMONSTER && !mon.player)
        {
            totalMonstersKilledThisSession++;
            killChainCounter = Min(killChainCounter + 2, 20);

            CVar enrageCv = CVar.FindCVar('adaptive_enrage_swarm');
            if (enrageCv && enrageCv.GetBool() && currentDifficulty >= 1.5)
            {
                BlockThingsIterator it = BlockThingsIterator.Create(mon, 512.0);
                while (it.Next())
                {
                    Actor ally = it.thing;
                    if (ally && ally.bISMONSTER && ally.health > 0 && ally != mon && !ally.player)
                    {
                        ally.bALWAYSFAST = true;
                        ally.Speed = ally.default.Speed * 1.5;
                    }
                }
            }

            CVar allowRespawnCv = CVar.FindCVar('adaptive_allow_respawn');
            if (allowRespawnCv && allowRespawnCv.GetBool())
            {
                DynamicRespawnItem item = new("DynamicRespawnItem");
                item.className = mon.GetClassName();
                item.spawnPos = mon.pos;

                CVar intensityCv = CVar.FindCVar('adaptive_respawn_intensity');
                double intensity = intensityCv ? intensityCv.GetFloat() : 1.0;
                double baseCooldownSeconds = 40.0 / (currentDifficulty * intensity);
                
                item.respawnTimer = int(35 * Max(5.0, baseCooldownSeconds));
                respawnQueue.Push(item);
            }
        }
    }

    void ProcessRespawnQueue(int playerHealth)
    {
        if (!IsEngineActive()) return;
        CVar floorCv = CVar.FindCVar('adaptive_respawn_floor_pct');
        int floorPct = floorCv ? floorCv.GetInt() : 40;
        if (playerHealth < floorPct) return;
        PlayerInfo p = players[consoleplayer];
        Vector3 pPos = (p && p.mo) ? p.mo.pos : (0, 0, 0);

        CVar minRangeCv = CVar.FindCVar('adaptive_respawn_min_range');
        double minRange = minRangeCv ? minRangeCv.GetFloat() : 512.0;

        for (int i = respawnQueue.Size() - 1; i >= 0; i--)
        {
            if (respawnQueue[i] == null)
            {
                respawnQueue.Delete(i);
                continue;
            }

            if (p && p.mo)
            {
                double playerDist = (respawnQueue[i].spawnPos - pPos).Length();
                if (playerDist < minRange) continue; 
            }

            respawnQueue[i].respawnTimer--;
            if (respawnQueue[i].respawnTimer <= 0)
            {
                Actor spawnedMon = Actor.Spawn(respawnQueue[i].className, respawnQueue[i].spawnPos, ALLOW_REPLACE);
                if (spawnedMon) 
                {
                    if (!spawnedMon.TestMobjLocation())
                    {
                        spawnedMon.Destroy();
                    }
                    else
                    {
                        spawnedMon.A_SpawnItemEx("TeleportFog");
                        
                        // Execute Uncapped Clone Multiplier Matrix explicitly marking it as a Respawn instance (true)
                        ExecuteCloningCore(spawnedMon, true);
                    }
                }
                respawnQueue.Delete(i);
            }
        }
    }

    void ProcessItemRecycleQueue()
    {
        PlayerInfo p = players[consoleplayer];
        Vector3 pPos = (p && p.mo) ? p.mo.pos : (0, 0, 0);
        CVar proxCv = CVar.FindCVar('adaptive_item_respawn_proximity_check');
        bool doProxCheck = proxCv ? proxCv.GetBool() : true;

        for (int i = 0; i < mapItemRegistry.Size(); i++)
        {
            if (mapItemRegistry[i].pickedUp)
            {
                if (doProxCheck && p && p.mo)
                {
                    double playerDist = (mapItemRegistry[i].spawnPos - pPos).Length();
                    if (playerDist < 128.0) continue;
                }

                mapItemRegistry[i].respawnTimer--;
                if (mapItemRegistry[i].respawnTimer <= 0)
                {
                    Actor drop = Actor.Spawn(mapItemRegistry[i].className, mapItemRegistry[i].spawnPos, ALLOW_REPLACE);
                    if (drop)
                    {
                        drop.A_SpawnItemEx("ItemFog");
                    }
                    mapItemRegistry[i].pickedUp = false;
                }
            }
        }
    }

    ui string GetDifficultyRankName()
    {
        if (currentDifficulty >= 10.0) return "ARMAGEDDON";
        if (currentDifficulty >= 5.0) return "SLAUGHTERFEST";
        if (currentDifficulty >= 2.0) return "NIGHTMARE";
        if (currentDifficulty >= 1.5) return "ULTRA-VIOLENCE";
        if (currentDifficulty >= 1.0) return "HURT ME PLENTY";
        if (currentDifficulty >= 0.7) return "HEY, NOT TOO ROUGH";
        return "I'M TOO YOUNG TO DIE";
    }

    override void RenderOverlay(RenderEvent e)
    {
        if (!IsEngineActive()) return;
        CVar showCv = CVar.FindCVar('adaptive_hud_show');
        if (showCv && !showCv.GetBool()) return;
        if (automapactive || !playeringame[consoleplayer]) return;
        
        PlayerInfo p = players[consoleplayer];
        if (!p || p.camera != p.mo) return;

        CVar xCv = CVar.FindCVar('adaptive_hud_x');
        CVar yCv = CVar.FindCVar('adaptive_hud_y');
        CVar scaleCv = CVar.FindCVar('adaptive_hud_scale');
        int posX = xCv ? xCv.GetInt() : 25;
        int posY = yCv ? yCv.GetInt() : 25;
        float hudScale = scaleCv ? scaleCv.GetFloat() : 1.0;

        int uiColor = Font.CR_GREEN;
        if (currentDifficulty >= 5.0)       uiColor = Font.CR_FIRE;
        else if (currentDifficulty >= 2.5)  uiColor = Font.CR_PURPLE;
        else if (currentDifficulty >= 1.7)  uiColor = Font.CR_RED;
        else if (currentDifficulty >= 1.3)  uiColor = Font.CR_GOLD;
        else if (currentDifficulty < 0.8)   uiColor = Font.CR_LIGHTBLUE;
        string rankTitle = GetDifficultyRankName();
        string diffLine = String.Format("STRESS PROFILE: %s (%.2f)", rankTitle, currentDifficulty);
        string killLine = String.Format("CAMPAIGN ELIMINATIONS: %d", totalMonstersKilledThisSession);
        int lineSpacing = int(22 * hudScale);

        Screen.DrawText("BigFont", uiColor, posX, posY, diffLine, 
            DTA_KeepRatio, true, DTA_ScaleX, hudScale, DTA_ScaleY, hudScale);
        Screen.DrawText("SmallFont", Font.CR_GREY, posX, posY + lineSpacing, killLine, 
            DTA_KeepRatio, true, DTA_ScaleX, hudScale, DTA_ScaleY, hudScale);
        CVar ammoProtCv = CVar.FindCVar('adaptive_ammo_protection');
        CVar aggChainsCv = CVar.FindCVar('adaptive_aggression_chains');

        string statusTag = "";
        if (ammoProtCv && ammoProtCv.GetBool() && isAmmoStarved) 
        {
            statusTag = "\cG[CRITICAL RESERVE: SPAWNS HALTED]";
        }
        else if (aggChainsCv && aggChainsCv.GetBool() && killChainCounter >= 8) 
        {
            statusTag = "\cH[KILL CHAIN ACTIVE: ENEMY DENSITY SURGING]";
        }

        if (statusTag != "")
        {
            Screen.DrawText("SmallFont", Font.CR_UNTRANSLATED, posX, posY + (lineSpacing * 2), statusTag, 
                DTA_KeepRatio, true, DTA_ScaleX, hudScale, DTA_ScaleY, hudScale);
        }
    }
